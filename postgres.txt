DROP TABLE dow;
DROP TABLE eth;

-- Create tables for raw data to be loaded into
CREATE TABLE dow (
date TEXT PRIMARY KEY,
dow_open FLOAT,
dow_close FLOAT	

);

CREATE TABLE eth (
date TEXT PRIMARY KEY,
eth_open FLOAT,
eth_close FLOAT
	
);


-- Joins tables
SELECT dow.date, dow.dow_open, dow.dow_close, eth.date, eth.eth_open, eth.eth_close
FROM eth
INNER JOIN dow
ON dow.date = eth.date;
